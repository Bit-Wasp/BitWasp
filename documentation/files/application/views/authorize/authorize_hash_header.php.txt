<script language="Javascript" type="text/javascript">
function make_hash() {

	// Generate a proof of work for the password by taking the
	// SHA512 hash of the submitted passwords (done 10 times) and 
	// submit that instead of the password.
	var hash0 = '';
	var pw0 = document.authorizeForm.password.value;
	var i = 0;
	
	hash0 = SHA512(pw0);
	
	for(i = 0; i < 9; i++) {
		hash0 = SHA512(hash0);
	}
	
	document.authorizeForm.password.value=hash0; 
	document.authorizeForm.submit();
}
</script>
